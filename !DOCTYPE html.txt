  <script>
    // Simulated single user balance stored locally
    let balance = parseFloat(localStorage.getItem("balance")) || 0;
    document.getElementById("balance").innerText = balance.toFixed(2);

    function updateDisplay() {
      document.getElementById("balance").innerText = balance.toFixed(2);
      localStorage.setItem("balance", balance);
    }

    function addBalance(amount) {
      balance += amount;
      updateDisplay();
      alert(`Added $${amount.toFixed(2)} to your balance.`);
    }

    function subtractBalance(amount) {
      if (balance >= amount) {
        balance -= amount;
        updateDisplay();
        alert(`Subtracted $${amount.toFixed(2)} from your balance.`);
      } else {
        alert("Not enough balance.");
      }
    }

    function showAddress() {
      const select = document.getElementById("crypto-select");
      const addressBox = document.getElementById("crypto-address");
      const value = select.value;

      if (value === "btc") {
        addressBox.innerText = "bc1qexamplebtcaddress000000000";
      } else if (value === "eth") {
        addressBox.innerText = "0xExampleEthereumAddress123456789";
      } else if (value === "usdt") {
        addressBox.innerText = "TRC20ExampleUSDTAddress99999999";
      }
    }

    function adminAddFunds() {
      const user = document.getElementById("admin-user").value.trim();
      const amount = parseFloat(document.getElementById("admin-amount").value);

      if (!user) {
        alert("Please enter user email or ID.");
        return;
      }
      if (isNaN(amount) || amount <= 0) {
        alert("Please enter a valid amount.");
        return;
      }

      // For demo, we only have one user (localStorage)
      // In a real app, you'd lookup user and update balance server-side

      balance += amount;
      updateDisplay();
      alert(`Admin added $${amount.toFixed(2)} to user "${user}".`);
      
      // Clear inputs
      document.getElementById("admin-user").value = "";
      document.getElementById("admin-amount").value = "";
    }
  </script>
